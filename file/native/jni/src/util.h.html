     14     static module_info_t get_module(const char *libname) {
     15         char buff[256];
     16         int len_libname = strlen(libname);
     17         uintptr_t start_offset = 0;
     18         uintptr_t end_offset = 0;
     19 
     20         auto file = fopen("/proc/self/smaps", "rt");
     21         if (file == NULL)
     22             return {0, 0};
     23 
     24         while (fgets(buff, sizeof buff, file) != NULL) {
     25             int len = strlen(buff);
     26             if (len > 0 && buff[len - 1] == '\n') {
     27                 buff[--len] = '\0';
     28             }
     29             if (len <= len_libname || memcmp(buff + len - len_libname, libname, len_libname)) {
     30                 continue;
     31             }
     32             size_t start, end, offset;
     33             char flags[4];
     34             if (sscanf(buff, "%zx-%zx %c%c%c%c %zx", &start, &end,
     35                        &flags[0], &flags[1], &flags[2], &flags[3], &offset) != 7) {
     36                 continue;
     37             }
     38 
     39             if (flags[0] != 'r' || flags[2] != 'x') {
     40                 continue;
     41             }
     42 
     43             if (start_offset == 0) {
     44                 start_offset = start;
     45             }
     46             end_offset = end;
     47         }
     48         fclose(file);
     49         if (start_offset == 0) {
     50             return {0, 0};
     51         }
     52         return { start_offset, end_offset - start_offset };
     53     }

     55     static uintptr_t find_signature(uintptr_t module_base, uintptr_t size, const std::string &pattern, int offset = 0) {
     56         std::vector<char> bytes;
     57         std::vector<char> mask;
     58         for (size_t i = 0; i < pattern.size(); i += 3) {
     59             if (pattern[i] == '?') {
     60                 bytes.push_back(0);
     61                 mask.push_back('?');
     62             } else {
     63                 bytes.push_back(std::stoi(pattern.substr(i, 2), nullptr, 16));
     64                 mask.push_back('x');
     65             }
     66         }
     67 
     68         for (size_t i = 0; i < size; i++) {
     69             bool found = true;
     70             for (size_t j = 0; j < bytes.size(); j++) {
     71                 if (mask[j] == '?' || bytes[j] == *(char *) (module_base + i + j)) {
     72                     continue;
     73                 }
     74                 found = false;
     75                 break;
     76             }
     77             if (found) {
     78                 return module_base + i + offset;
     79             }
     80         }
     81         return 0;
     82     }
