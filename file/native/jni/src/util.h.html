     14     static module_info_t get_module(const char *libname) {
     15         char buff[256];
     16         int len_libname = strlen(libname);
     17         uintptr_t start_offset = 0;
     18         uintptr_t end_offset = 0;
     19 
     20         auto file = fopen("/proc/self/smaps", "rt");
     21         if (file == NULL)
     22             return {0, 0};
     23 
     24         while (fgets(buff, sizeof buff, file) != NULL) {
     25             int len = strlen(buff);
     26             if (len > 0 && buff[len - 1] == '\n') {
     27                 buff[--len] = '\0';
     28             }
     29             if (len <= len_libname || memcmp(buff + len - len_libname, libname, len_libname)) {
     30                 continue;
     31             }
     32             size_t start, end, offset;
     33             char flags[4];
     34             if (sscanf(buff, "%zx-%zx %c%c%c%c %zx", &start, &end,
     35                        &flags[0], &flags[1], &flags[2], &flags[3], &offset) != 7) {
     36                 continue;
     37             }
     38 
     39             if (flags[0] != 'r' || flags[2] != 'x') {
     40                 continue;
     41             }
     42 
     43             if (start_offset == 0) {
     44                 start_offset = start;
     45             }
     46             end_offset = end;
     47         }
     48         fclose(file);
     49         if (start_offset == 0) {
     50             return {0, 0};
     51         }
     52         return { start_offset, end_offset - start_offset };
     53     }

     55     static void remap_sections(std::function<bool(const std::string &, size_t)> filter, bool remove_read_permission) {
     56         char buff[256];
     57         auto maps = fopen("/proc/self/maps", "rt");
     58 
     59         while (fgets(buff, sizeof buff, maps) != NULL) {
     60             int len = strlen(buff);
     61             if (len > 0 && buff[len - 1] == '\n') buff[--len] = '\0';
     62 
     63             size_t start, end, offset;
     64             char flags[4];
     65 
     66             if (sscanf(buff, "%zx-%zx %c%c%c%c %zx", &start, &end,
     67                        &flags[0], &flags[1], &flags[2], &flags[3], &offset) != 7) continue;
     68 
     69             if (!filter(buff, end - start)) continue;
     70 
     71             auto section_size = end - start;
     72             auto section_ptr = mmap(0, section_size, PROT_READ | PROT_EXEC | PROT_WRITE, MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
     73 
     74             if (section_ptr == MAP_FAILED) {
     75                 LOGE("mmap failed: %s", strerror(errno));
     76                 break;
     77             }
     78 
     79             memcpy(section_ptr, (void *)start, section_size);
     80 
     81             if (mremap(section_ptr, section_size, section_size, MREMAP_MAYMOVE | MREMAP_FIXED, start) == MAP_FAILED) {
     82                 LOGE("mremap failed: %s", strerror(errno));
     83                 break;
     84             }
     85 
     86             auto new_prot = (flags[0] == 'r' ? PROT_READ : 0) | (flags[1] == 'w' ? PROT_WRITE : 0) | (flags[2] == 'x' ? PROT_EXEC : 0);
     87             if (remove_read_permission && flags[0] == 'r' && flags[2] == 'x') {
     88                 new_prot &= ~PROT_READ;
     89             }
     90             mprotect((void *)start, section_size, new_prot);
     91         }
     92         fclose(maps);
     93     }

     95     static uintptr_t find_signature(uintptr_t module_base, uintptr_t size, const std::string &pattern, int offset = 0) {
     96         std::vector<char> bytes;
     97         std::vector<char> mask;
     98         for (size_t i = 0; i < pattern.size(); i += 3) {
     99             if (pattern[i] == '?') {
    100                 bytes.push_back(0);
    101                 mask.push_back('?');
    102             } else {
    103                 bytes.push_back(std::stoi(pattern.substr(i, 2), nullptr, 16));
    104                 mask.push_back('x');
    105             }
    106         }
    107 
    108         for (size_t i = 0; i < size; i++) {
    109             bool found = true;
    110             for (size_t j = 0; j < bytes.size(); j++) {
    111                 if (mask[j] == '?' || bytes[j] == *(char *) (module_base + i + j)) {
    112                     continue;
    113                 }
    114                 found = false;
    115                 break;
    116             }
    117             if (found) {
    118                 return module_base + i + offset;
    119             }
    120         }
    121         return 0;
    122     }
