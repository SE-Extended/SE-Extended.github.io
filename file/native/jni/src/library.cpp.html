     17 bool JNICALL init(JNIEnv *env, jobject clazz) {
     18     LOGD("Initializing native");
     19     using namespace common;
     20     util::remap_sections([](const std::string &line, size_t size) {
     21         return line.find(BUILD_PACKAGE) != std::string::npos;
     22     }, native_config->remap_executable);
     23 
     24     native_lib_object = env->NewGlobalRef(clazz);
     25     client_module = util::get_module("libclient.so");
     26 
     27     if (client_module.base == 0) {
     28         LOGD("libclient.so not found, trying split_config");
     29         client_module = util::get_module(("split_config." + std::string(ARM64 ? "arm64_v8a" : "armeabi-v7a") + ".apk").c_str());
     30         if (client_module.base == 0) {
     31             LOGE("can't find split_config!");
     32             return false;
     33         }
     34     }
     35 
     36     LOGD("client_module offset=0x%lx, size=0x%zx", client_module.base, client_module.size);
     37 
     38     auto threads = std::vector<std::thread>();
     39 
     40     #define RUN(body) threads.push_back(std::thread([&] { body; }))
     41 
     42     RUN(UnaryCallHook::init(env));
     43     RUN(FstatHook::init());
     44     RUN(SqliteMutexHook::init());
     45     RUN(DuplexHook::init(env));
     46     if (common::native_config->custom_emoji_font_path[0] != 0) {
     47         RUN(CustomEmojiFont::init());
     48     }
     49     if (common::native_config->composer_hooks) {
     50         RUN(ComposerHook::init());
     51     }
     52 
     53     for (auto &thread : threads) {
     54         thread.join();
     55     }
     56 
     57     LOGD("Native initialized");
     58     return true;
     59 }

     61 void JNICALL load_config(JNIEnv *env, jobject, jobject config_object) {
     62     auto native_config_clazz = env->GetObjectClass(config_object);
     63 #define GET_CONFIG_BOOL(name) env->GetBooleanField(config_object, env->GetFieldID(native_config_clazz, name, "Z"))
     64     auto native_config = common::native_config;
     65 
     66     native_config->disable_bitmoji = GET_CONFIG_BOOL("disableBitmoji");
     67     native_config->disable_metrics = GET_CONFIG_BOOL("disableMetrics");
     68     native_config->composer_hooks = GET_CONFIG_BOOL("composerHooks");
     69     native_config->remap_executable = GET_CONFIG_BOOL("remapExecutable");
     70 
     71     memset(native_config->custom_emoji_font_path, 0, sizeof(native_config->custom_emoji_font_path));
     72     auto custom_emoji_font_path = env->GetObjectField(config_object, env->GetFieldID(native_config_clazz, "customEmojiFontPath", "Ljava/lang/String;"));
     73     if (custom_emoji_font_path != nullptr) {
     74         auto custom_emoji_font_path_str = env->GetStringUTFChars((jstring) custom_emoji_font_path, nullptr);
     75         strncpy(native_config->custom_emoji_font_path, custom_emoji_font_path_str, sizeof(native_config->custom_emoji_font_path));
     76         env->ReleaseStringUTFChars((jstring) custom_emoji_font_path, custom_emoji_font_path_str);
     77     }

     80 void JNICALL lock_database(JNIEnv *env, jobject, jstring database_name, jobject runnable) {
     81     auto database_name_str = env->GetStringUTFChars(database_name, nullptr);
     82     auto mutex = SqliteMutexHook::mutex_map[database_name_str];
     83     env->ReleaseStringUTFChars(database_name, database_name_str);
     84 
     85     if (mutex != nullptr) {
     86         auto lock_result = pthread_mutex_lock(&mutex->mutex);
     87         if (lock_result != 0) {
     88             LOGE("pthread_mutex_lock failed: %d", lock_result);
     89             return;
     90         }
     91     }
     92 
     93     env->CallVoidMethod(runnable, env->GetMethodID(env->GetObjectClass(runnable), "run", "()V"));
     94 
     95     if (mutex != nullptr) {
     96         pthread_mutex_unlock(&mutex->mutex);
     97     }
     98 }

    100 void JNICALL hide_anonymous_dex_files(JNIEnv *, jobject) {
    101     util::remap_sections([](const std::string &line, size_t size) {
    102         return (
    103             (common::native_config->remap_executable && size == PAGE_SIZE && line.find("r-xp 00000000 00") != std::string::npos && line.find("[v") == std::string::npos) ||
    104             line.find("dalvik-DEX") != std::string::npos ||
    105             line.find("dalvik-classes") != std::string::npos
    106         );
    107     }, common::native_config->remap_executable);
    108 }
