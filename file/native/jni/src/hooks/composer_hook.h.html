     98     HOOK_DEF(JSValue, js_eval, void* instance, JSContext *ctx, void* this_obj, char *input, uintptr_t input_len, const char *filename, unsigned int flags, unsigned int scope_idx) {
     99         if (global_instance == nullptr || global_ctx == nullptr) {
    100             global_instance = instance;
    101             global_ctx = ctx;
    102 
    103             if (composer_loader != nullptr) {
    104                 LOGD("Injecting composer loader");
    105                 composer_loader->resize(composer_loader->size() + input_len);
    106                 memcpy((void*) (composer_loader->c_str() + composer_loader->size() - input_len), input, input_len);
    107 
    108                 input = (char*) composer_loader->c_str();
    109                 input_len = composer_loader->size();
    110             }
    111         } else {
    112             if (composer_loader != nullptr) {
    113                 delete composer_loader;
    114                 composer_loader = nullptr;
    115             }
    116         }
    117         return js_eval_original(instance, ctx, this_obj, input, input_len, filename, flags, scope_idx);
    118     }

    129     jstring composerEval(JNIEnv *env, jobject, jstring script) {
    130         if (!ARM64) return env->NewStringUTF("Architecture not supported");
    131         if (global_instance == 0 || global_ctx == nullptr) {
    132             return env->NewStringUTF("Composer not ready");
    133         }
    134 
    135         auto script_str = env->GetStringUTFChars(script, nullptr);
    136         auto length = env->GetStringUTFLength(script);
    137         auto jsvalue = js_eval_original(global_instance, global_ctx, (void*) &global_ctx->global_obj, (char *) script_str, length, "<eval>", 0, 0);
    138         env->ReleaseStringUTFChars(script, script_str);
    139 
    140         if (jsvalue.tag == JS_TAG_STRING) {
    141             auto str = (JSString *) jsvalue.u.ptr;
    142             return env->NewStringUTF((const char *) str->u.str8);
    143         }
    144 
    145         std::string result;
    146         switch (jsvalue.tag) {
    147             case JS_TAG_INT:
    148                 result = std::to_string(jsvalue.u.int32);
    149                 break;
    150             case JS_TAG_BOOL:
    151                 result = jsvalue.u.int32 ? "true" : "false";
    152                 break;
    153             case JS_TAG_NULL:
    154                 result = "null";
    155                 break;
    156             case JS_TAG_UNDEFINED:
    157                 result = "undefined";
    158                 break;
    159             case JS_TAG_OBJECT:
    160                 result = "[object Object]";
    161                 break;
    162             case JS_TAG_EXCEPTION:
    163                 result = "Failed to evaluate script";
    164                 break;
    165             case JS_TAG_FLOAT64:
    166                 result = std::to_string(jsvalue.u.float64);
    167                 break;
    168             default:
    169                 result = "[unknown tag " + std::to_string(jsvalue.tag) + "]";
    170                 break;
    171         }
    172 
    173         return env->NewStringUTF(result.c_str());
    174     }
