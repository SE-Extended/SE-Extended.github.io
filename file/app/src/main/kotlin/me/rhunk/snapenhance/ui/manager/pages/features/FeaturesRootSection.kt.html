    654     @Composable
    655     private fun PropertiesView(
    656         properties: List<PropertyPair<*>>
    657     ) {
    658         Scaffold(
    659             modifier = Modifier.fillMaxSize(),
    660             content = { innerPadding ->
    661                 LazyColumn(
    662                     modifier = Modifier
    663                         .fillMaxHeight()
    664                         .padding(innerPadding),
    665                     //save button space
    666                     contentPadding = PaddingValues(top = 10.dp, bottom = 110.dp),
    667                     verticalArrangement = Arrangement.Top
    668                 ) {
    669                     items(properties) {
    670                         PropertyCard(it)
    671                     }
    672                 }
    673             }
    674         )
    675     }
    676 
    677     override val floatingActionButton: @Composable () -> Unit = {
    678         fun saveConfig() {
    679             context.coroutineScope.launch(Dispatchers.IO) {
    680                 context.config.writeConfig()
    681                 context.log.verbose("saved config!")
    682             }
    683         }
    684 
    685         OnLifecycleEvent { _, event ->
    686             if (event == Lifecycle.Event.ON_PAUSE || event == Lifecycle.Event.ON_STOP) {
    687                 saveConfig()
    688             }
    689         }
    690 
    691         DisposableEffect(Unit) {
    692             onDispose {
    693                 saveConfig()
    694             }
    695         }
    696     }

    701         configContainer: ConfigContainer

    704             configContainer.properties.map { PropertyPair(it.key, it.value) }
    705         })
