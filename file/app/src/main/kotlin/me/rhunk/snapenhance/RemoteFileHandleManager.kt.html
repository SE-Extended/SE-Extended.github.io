     50     private val context: RemoteSideContext

     52     private val userImportFolder = File(context.androidContext.filesDir, "user_imports").apply {

     56     override fun getFileHandle(scope: String, name: String): FileHandle? {
     57         val fileHandleScope = FileHandleScope.fromValue(scope) ?: run {
     58             context.log.error("invalid file handle scope: $scope", "FileHandleManager")
     59             return null
     60         }
     61         when (fileHandleScope) {
     62             FileHandleScope.INTERNAL -> {
     63                 val fileHandleType = InternalFileHandleType.fromValue(name) ?: run {
     64                     context.log.error("invalid file handle name: $name", "FileHandleManager")
     65                     return null
     66                 }
     67 
     68                 return LocalFileHandle(
     69                     fileHandleType.resolve(context.androidContext)
     70                 )
     71             }
     72             FileHandleScope.LOCALE -> {
     73                 val foundLocale = context.androidContext.resources.assets.list("lang")?.firstOrNull {
     74                     it.startsWith(name)
     75                 }?.substringBefore(".") ?: return null
     76 
     77                 if (name == LocaleWrapper.DEFAULT_LOCALE) {
     78                     return AssetFileHandle(
     79                         context,
     80                         "lang/${LocaleWrapper.DEFAULT_LOCALE}.json"
     81                     )
     82                 }
     83 
     84                 return AssetFileHandle(
     85                     context,
     86                     "lang/$foundLocale.json"
     87                 )
     88             }
     89             FileHandleScope.USER_IMPORT -> {
     90                 return LocalFileHandle(
     91                     File(userImportFolder, name.substringAfterLast("/"))
     92                 )
     93             }
     94             FileHandleScope.COMPOSER -> {
     95                 return AssetFileHandle(
     96                     context,
     97                     "composer/${name.substringAfterLast("/")}"
     98                 )
     99             }
    100             else -> return null
    101         }
    102     }

    104     fun getStoredFiles(filter: ((File) -> Boolean)? = null): List<File> {
    105         return userImportFolder.listFiles()?.toList()?.let { files ->
    106             filter?.let { files.filter(it) } ?: files
    107         }?.sortedBy { -it.lastModified() } ?: emptyList()
    108     }

    110     fun getFileInfo(name: String): Pair<Long, Long>? {
    111         return runCatching {
    112             val file = File(userImportFolder, name)
    113             file.length() to file.lastModified()
    114         }.onFailure {
    115             context.log.error("Failed to get file info: ${it.message}", it)
    116         }.getOrNull()
    117     }

    119     fun importFile(name: String, block: OutputStream.() -> Unit): Boolean {
    120         return runCatching {
    121             val file = File(userImportFolder, name)
    122             file.outputStream().use(block)
    123             true
    124         }.onFailure {
    125             context.log.error("Failed to import file: ${it.message}", it)
    126         }.getOrDefault(false)
    127     }
    128 
    129     fun deleteFile(name: String): Boolean {
    130         return runCatching {
    131             File(userImportFolder, name).delete()
    132         }.onFailure {
    133             context.log.error("Failed to delete file: ${it.message}", it)
    134         }.isSuccess
    135     }
