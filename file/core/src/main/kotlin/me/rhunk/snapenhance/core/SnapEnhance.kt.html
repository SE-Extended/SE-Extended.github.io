     48     private fun hookMainActivity(methodName: String, stage: HookStage = HookStage.AFTER, block: Activity.(param: HookAdapter) -> Unit) {
     49         Activity::class.java.hook(methodName, stage, { isBridgeInitialized }) { param ->
     50             val activity = param.thisObject() as Activity
     51             if (!activity.packageName.equals(Constants.SNAPCHAT_PACKAGE_NAME)) return@hook
     52             block(activity, param)
     53         }
     54     }

     56     fun init(context: Context) {
     57         appContext = ModContext(
     58             androidContext = context.also { classLoader = it.classLoader }
     59         )
     60         appContext.apply {
     61             bridgeClient = BridgeClient(this)
     62             initConfigListener()
     63             bridgeClient.addOnConnectedCallback {
     64                 bridgeClient.registerMessagingBridge(messagingBridge)
     65                 coroutineScope.launch {
     66                     runCatching {
     67                         syncRemote()
     68                     }.onFailure {
     69                         log.error("Failed to sync remote", it)
     70                     }
     71                 }
     72             }
     73         }
     74 
     75         runBlocking {
     76             var throwable: Throwable? = null
     77             val canLoad = appContext.bridgeClient.connect { throwable = it }
     78             if (canLoad == null) {
     79                 InAppOverlay.showCrashOverlay(
     80                     buildString {
     81                         append("Snapchat timed out while trying to connect to SE Extended\n\n")
     82                         append("Make sure you:\n")
     83                         append(" - Have installed the latest SE Extended version (https://github.com/bocajthomas/SE-Extended)\n")
     84                         append(" - Disabled battery optimizations\n")
     85                         append(" - Excluded SE Extended and Snapchat in HideMyApplist")
     86                     },
     87                     throwable
     88                 )
     89                 appContext.logCritical("Cannot connect to the SE Extended app")
     90                 return@runBlocking
     91             }
     92             if (!canLoad) exitProcess(1)
     93             runCatching {
     94                 LSPatchUpdater.onBridgeConnected(appContext)
     95             }.onFailure {
     96                 appContext.log.error("Failed to init LSPatchUpdater", it)
     97             }
     98             jetpackComposeResourceHook()
     99             runCatching {
    100                 measureTimeMillis {
    101                     init(this)
    102                 }.also {
    103                     appContext.log.verbose("init took ${it}ms")
    104                 }
    105 
    106                 hookMainActivity("onPostCreate") {
    107                     appContext.mainActivity = this
    108                     if (!appContext.mappings.isMappingsLoaded) return@hookMainActivity
    109                     appContext.isMainActivityPaused = false
    110                     onActivityCreate(this)
    111                     appContext.actionManager.onNewIntent(intent)
    112                 }
    113 
    114                 hookMainActivity("onPause") {
    115                     appContext.bridgeClient.closeOverlay()
    116                     appContext.isMainActivityPaused = true
    117                 }
    118 
    119                 hookMainActivity("onNewIntent") { param ->
    120                     appContext.actionManager.onNewIntent(param.argNullable(0))
    121                 }
    122 
    123                 hookMainActivity("onResume") {
    124                     if (appContext.isMainActivityPaused.also {
    125                             appContext.isMainActivityPaused = false
    126                         }) {
    127                         appContext.reloadConfig()
    128                         appContext.executeAsync {
    129                             syncRemote()
    130                         }
    131                     }
    132                 }
    133             }.onSuccess {
    134                 isBridgeInitialized = true
    135             }.onFailure {
    136                 appContext.logCritical("Failed to initialize bridge", it)
    137                 InAppOverlay.showCrashOverlay("SE Extended failed to initialize. Please check logs for more details.", it)
    138             }
    139         }
    140     }

    142     private fun init(scope: CoroutineScope) {
    143         with(appContext) {
    144             Thread::class.java.hook("dispatchUncaughtException", HookStage.BEFORE) { param ->

    151             reloadConfig()
    152             initWidgetListener()

    157             }

    159             mappings.init(androidContext)
    160             database.init()
    161             eventDispatcher.init()
    162             //if mappings aren't loaded, we can't initialize features

    164             features.init()

    166             scriptRuntime.eachModule { callFunction("module.onSnapApplicationLoad", androidContext) }
    167         }

    189     private fun initNative() {
    190         val lateInit = appContext.native.initOnce {
    191             nativeUnaryCallCallback = { request ->
    192                 appContext.event.post(NativeUnaryCallEvent(request.uri, request.buffer)) {
    193                     request.buffer = buffer
    194                     request.canceled = canceled
    195                 }
    196             }
    197             appContext.reloadNativeConfig()
    198         }
    199 
    200         if (appContext.bridgeClient.getDebugProp("disable_sif", "false") != "true") {
    201             runCatching {
    202                 appContext.native.loadSharedLibrary(
    203                     appContext.fileHandlerManager.getFileHandle(FileHandleScope.INTERNAL.key, InternalFileHandleType.SIF.key)
    204                         .toWrapper()
    205                         .readBytes()
    206                         .takeIf {
    207                             it.isNotEmpty()
    208                         } ?: throw IllegalStateException("buffer is empty")
    209                 )
    210                 appContext.log.verbose("loaded sif")
    211             }.onFailure {
    212                 safeMode = true
    213                 appContext.log.error("Failed to load sif", it)
    214             }
    215         } else {
    216             appContext.log.warn("sif is disabled")
    217         }
    218 
    219         Runtime::class.java.declaredMethods.first {
    220             it.name == "loadLibrary0" && it.parameterTypes.contentEquals(
    221                 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) arrayOf(Class::class.java, String::class.java)
    222                 else arrayOf(ClassLoader::class.java, String::class.java)
    223             )
    224         }.apply {
    225             if (safeMode) {
    226                 hook(HookStage.BEFORE) { param ->
    227                     if (param.arg<String>(1) != "scplugin") return@hook
    228                     param.setResult(null)
    229                     appContext.log.warn("Can't load scplugin in safe mode")
    230                     runCatching {
    231                         Thread.sleep(Long.MAX_VALUE)
    232                     }.onFailure {
    233                         appContext.log.error(it)
    234                     }
    235                     exitProcess(1)
    236                 }
    237             }
    238 
    239             lateinit var unhook: () -> Unit
    240             hook(HookStage.AFTER) { param ->
    241                 if (param.arg<String>(1) != "client") return@hook
    242                 unhook()
    243                 appContext.log.verbose("libclient lateInit")
    244                 lateInit()
    245             }.also { unhook = { it.unhook() } }
    246         }
    247     }

    290     private fun initWidgetListener() {
    291         appContext.event.subscribe(SnapWidgetBroadcastReceiveEvent::class) { event ->
    292             if (event.action != ReceiversConfig.BRIDGE_SYNC_ACTION) return@subscribe
    293             event.canceled = true
    294             val feedEntries = appContext.database.getFeedEntries(Int.MAX_VALUE)
    295 
    296             val groups = feedEntries.filter { it.friendUserId == null }.map {
    297                 MessagingGroupInfo(
    298                     it.key!!,
    299                     it.feedDisplayName!!,
    300                     it.participantsSize
    301                 )
    302             }
    303 
    304             val friends = feedEntries.filter { it.friendUserId != null }.map {
    305                 MessagingFriendInfo(
    306                     it.friendUserId!!,
    307                     appContext.database.getConversationLinkFromUserId(it.friendUserId!!)?.clientConversationId,
    308                     it.friendDisplayName,
    309                     it.friendDisplayUsername!!.split("|")[1],
    310                     it.bitmojiAvatarId,
    311                     it.bitmojiSelfieId,
    312                     streaks = null
    313                 )
    314             }
    315 
    316             appContext.bridgeClient.passGroupsAndFriends(groups, friends)
    317         }
    318     }

    353     private fun jetpackComposeResourceHook() {
    354         fun strings(vararg classes: KClass<*>): Map<Int, String> {
    355             return classes.fold(mapOf()) { map, clazz ->
    356                 map + clazz.java.fields.filter {
    357                     java.lang.reflect.Modifier.isStatic(it.modifiers) && it.type == Int::class.javaPrimitiveType
    358                 }.associate { it.getInt(null) to it.name }
    359             }
    360         }
    361         val stringResources = strings(androidx.compose.material3.R.string::class, androidx.compose.ui.R.string::class)
    362         Resources::class.java.getMethod("getString", Int::class.javaPrimitiveType).hook(HookStage.BEFORE) { param ->
    363             val key = param.arg<Int>(0)
    364             val name = stringResources[key]?.replaceFirst("m3c_", "") ?: return@hook
    365             param.setResult(appContext.translation.getOrNull("material3_strings.${name}") ?: "")
    366         }
    367     }
