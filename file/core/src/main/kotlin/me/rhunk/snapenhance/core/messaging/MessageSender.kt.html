    102     private fun internalSendMessage(conversations: List<SnapUUID>, localMessageContentTemplate: String, callback: Any) {
    103         val sendMessageWithContentMethod = context.classCache.conversationManager.declaredMethods.first { it.name == "sendMessageWithContent" }
    104 
    105         val localMessageContent = context.gson.fromJson(localMessageContentTemplate, context.classCache.localMessageContent)
    106         val messageDestinations = MessageDestinations(AbstractWrapper.newEmptyInstance(context.classCache.messageDestinations)).also {
    107             it.conversations = conversations.toCollection(ArrayList())
    108             it.mPhoneNumbers = arrayListOf<Any>()
    109             it.stories = arrayListOf<Any>()
    110         }
    111 
    112         sendMessageWithContentMethod.invoke(context.feature(Messaging::class).conversationManager?.instanceNonNull(), messageDestinations.instanceNonNull(), localMessageContent, callback)
    113     }

    115     fun sendChatMessage(conversations: List<SnapUUID>, message: String, onError: (Any) -> Unit = {}, onSuccess: () -> Unit = {}) {
    116         internalSendMessage(conversations, createLocalMessageContentTemplate(ContentType.CHAT, ProtoWriter().apply {
    117             from(2) {
