    294     private fun tryDecryptMessage(senderId: String, clientMessageId: Long, conversationId: String, contentType: ContentType, messageBuffer: ByteArray): Pair<ContentType, ByteArray> {
    295         if (contentType != ContentType.STATUS && decryptedMessageCache.containsKey(clientMessageId)) {
    296             return decryptedMessageCache[clientMessageId]!!
    297         }
    298 
    299         val reader = ProtoReader(messageBuffer)
    300         var outputBuffer = messageBuffer
    301         var outputContentType = fixContentType(contentType, reader) ?: contentType
    302         val conversationParticipants by lazy {
    303             getE2EParticipants(conversationId)
    304         }

    306         fun setDecryptedMessage(buffer: ByteArray) {
    307             outputBuffer = buffer
    308             outputContentType = fixContentType(outputContentType, ProtoReader(buffer)) ?: outputContentType
    309             decryptedMessageCache[clientMessageId] = outputContentType to buffer
    310             encryptedMessages.add(clientMessageId)
    311         }

    313         fun setWarningMessage() {
    314             encryptedMessages.add(clientMessageId)
    315             outputContentType = ContentType.CHAT
    316             outputBuffer = ProtoWriter().apply {
    317                 from(2) {
    318                     addString(1, "Failed to decrypt message, id=$clientMessageId. Check logs for more details.")
    319                 }
    320             }.toByteArray()
    321         }

    323         fun replaceMessageText(text: String) {
    324             outputBuffer = ProtoWriter().apply {
    325                 from(2) {
    326                     addString(1, text)
    327                 }
    328             }.toByteArray()
    329         }
    330 
    331         // decrypt messages
    332         reader.followPath(2, 1) {
    333             val messageTypeId = getVarInt(1)?.toInt() ?: return@followPath
    334             val isMe = context.database.myUserId == senderId
    335 
    336             if (messageTypeId == ENCRYPTED_MESSAGE_ID) {
    337                 runCatching {
    338                     eachBuffer(2) {
    339                         if (decryptedMessageCache.containsKey(clientMessageId)) return@eachBuffer
    340 
    341                         val participantIdHash = getByteArray(1) ?: return@eachBuffer
    342                         val iv = getByteArray(2) ?: return@eachBuffer
    343                         val ciphertext = getByteArray(3) ?: return@eachBuffer
    344 
    345                         if (isMe) {
    346                             if (conversationParticipants.isEmpty()) return@eachBuffer
    347                             val participantId = conversationParticipants.firstOrNull { participantIdHash.contentEquals(hashParticipantId(it, iv)) } ?: return@eachBuffer
    348                             setDecryptedMessage(e2eeInterface.decryptMessage(participantId, ciphertext, iv) ?: run {
    349                                 context.log.warn("Failed to decrypt message for participant $participantId")
    350                                 setWarningMessage()
    351                                 return@eachBuffer
    352                             })
    353                             return@eachBuffer
    354                         }
    355 
    356                         if (!participantIdHash.contentEquals(hashParticipantId(context.database.myUserId, iv))) return@eachBuffer
    357 
    358                         setDecryptedMessage(e2eeInterface.decryptMessage(senderId, ciphertext, iv)?: run {
    359                             setWarningMessage()
    360                             return@eachBuffer
    361                         })
    362                     }
    363                 }.onFailure {
    364                     context.log.error("Failed to decrypt message id: $clientMessageId", it)
    365                     setWarningMessage()
    366                 }
    367 
    368                 return@followPath
    369             }
    370 
    371             val payload = getByteArray(2, 2) ?: return@followPath
    372 
    373             if (senderId == context.database.myUserId) {
    374                 when (messageTypeId) {
    375                     REQUEST_PK_MESSAGE_ID -> {
    376                         replaceMessageText("[${translation["outgoing_pk_message"]}]")
    377                     }
    378                     RESPONSE_SK_MESSAGE_ID -> {
    379                         replaceMessageText("[${translation["outgoing_secret_message"]}]")
    380                     }
    381                 }
    382                 return@followPath
    383             }
    384 
    385             when (messageTypeId) {
    386                 REQUEST_PK_MESSAGE_ID -> {
    387                     pkRequests[clientMessageId] = payload
    388                     replaceMessageText(translation["incoming_pk_message"])
    389                 }
    390                 RESPONSE_SK_MESSAGE_ID -> {
    391                     secretResponses[clientMessageId] = payload
    392                     replaceMessageText(translation["incoming_secret_message"])
    393                 }
    394             }
    395         }
    396 
    397         return outputContentType to outputBuffer
    398     }

    400     private fun messageHook(conversationId: String, messageId: Long, senderId: String, messageContent: MessageContent, committed: Boolean) {
    401         val (contentType, buffer) = tryDecryptMessage(senderId, messageId, conversationId, messageContent.contentType ?: ContentType.CHAT, messageContent.content!!)
    402         messageContent.contentType = contentType
    403         messageContent.content = buffer
    404         // remove messages currently being sent from the cache
    405         if (!committed) {
    406             decryptedMessageCache.remove(messageId)
    407             encryptedMessages.remove(messageId)
    408         }
    409     }

    411     private fun ProtoWriter.writeEncryptedMessage(
    412         participantsIds: List<String>,
    413         messageContent: ByteArray,
    414     ) {
    415         from(2) {
    416             from(1) {
    417                 addVarInt(1, ENCRYPTED_MESSAGE_ID)
    418                 participantsIds.forEach { participantId ->
    419                     val encryptedMessage = e2eeInterface.encryptMessage(participantId,
    420                         messageContent
    421                     ) ?: run {
    422                         throw Exception("Failed to encrypt message for participant $participantId")
    423                     }
    424                     context.log.debug("encrypted message size = ${encryptedMessage.ciphertext.size}")
    425                     from(2) {
    426                         // participantId is hashed with iv to prevent leaking it when sending to multiple conversations
    427                         addBuffer(1, hashParticipantId(participantId, encryptedMessage.iv))
    428                         addBuffer(2, encryptedMessage.iv)
    429                         addBuffer(3, encryptedMessage.ciphertext)
    430                     }
    431                 }
    432                 if (ContentType.fromMessageContainer(ProtoReader(messageContent)) == ContentType.SNAP) {
    433                     addVarInt(5, 1)
    434                 }
    435             }
    436         }
    437     }

    475             val messageContent = event.messageContent
    476             val destinations = event.destinations

    497                             writeEncryptedMessage(e2eeConversations.map { getE2EParticipants(it) }.flatten().distinct(), messageContent.content!!)
    498                         }.toByteArray()
    499                     }.onFailure {
    500                         context.log.error("Failed to encrypt message", it)
    501                         context.longToast(translation["encryption_failed_toast"])
    502                     }
    503                 }

    507                 }
